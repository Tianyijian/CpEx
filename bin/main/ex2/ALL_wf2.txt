P -> D P | S P | ε | B

D -> X id ;
X -> integer | real

S -> id = E ;
E -> E + T | T
T -> T * F | F
F -> ( E ) 
F -> digit
F -> id

S -> if B then M S N else M S
S -> while M B do M S
B -> B or M A | A
A -> A and M C | C
C -> not C | ( B ) | E relop E | true | false    
relop -> < | <= | == | != | > | >= 
M -> ε
N -> ε
